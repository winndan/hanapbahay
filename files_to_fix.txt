@rt("/user")
async def get_user_dashboard(req):
    """Retrieve user session and display the user dashboard."""
    
    print("üîç Request Headers:", req.headers)
    print("üîç Request Cookies:", req.cookies)

    auth_header = req.headers.get("Authorization")
    token = None

    if auth_header and auth_header.startswith("Bearer "):
        token = auth_header.split("Bearer ")[1].strip()
    else:
        token = req.cookies.get("auth_token")  # Fallback to cookies

    if not token:
        print("üö® No token found! Redirecting to signin.")
        return Redirect("/signin")

    try:
        # ‚úÖ Fix: Remove `await`, as `supabase.auth.get_user(token)` is NOT async
        response = supabase.auth.get_user(token)

        if not response or not response.user:
            print("üö® Invalid token! Redirecting to signin.")
            return Redirect("/signin")

        user_data = response.user
        display_name = user_data.user_metadata.get("display_name", "User")

        return user_page(display_name)

    except Exception as e:
        print("‚ùå Error fetching user data from Supabase:", e)
        return Redirect("/signin")

@rt("/api/signin", methods="post")
async def post_signin_account(email: str, password: str):
    """Handles user login and sets auth token."""
    try:
        print("üîç Step 1: Received login request for email:", email)

        if supabase is None:
            print("üö® ERROR: Supabase client is not initialized!")
            return Script("window.location.href='/signin';")

        login_response = supabase.auth.sign_in_with_password({
            "email": email,
            "password": password
        })

        print("üîç Step 2: Login Response:", login_response)

        if not login_response or not getattr(login_response, "user", None):
            print("üö® ERROR: Invalid credentials!")
            return Script("window.location.href='/signin';")

        # ‚úÖ Extract user session data
        session = getattr(login_response, "session", None)
        if session is None:
            print("üö® ERROR: No session returned from Supabase!")
            return Script("window.location.href='/signin';")

        access_token = getattr(session, "access_token", None)
        if not access_token:
            print("üö® ERROR: No access token returned!")
            return Script("window.location.href='/signin';")

        print("üü¢ Step 3: User authenticated successfully.")

        # ‚úÖ Use JavaScript to redirect since HTMX doesn't follow `303 Redirect`
        response = Response(
            Script(f"""
                document.cookie = 'auth_token={access_token}; path=/; Secure;';
                window.location.href='/user';
            """),
            status_code=200
        )

        print("üü¢ Step 4: Redirecting to dashboard.")
        return response

    except Exception as e:
        print("‚ùå Exception in signin_account:", e)
        return Script("window.location.href='/signin';")

from fasthtml.common import *  # ‚úÖ Already imported
from monsterui.all import *
from db_connect import supabase

async def signin_account(email: str, password: str):
    try:
        print("üîç Step 1: Received login request for email:", email)

        # ‚úÖ Ensure Supabase client exists
        if supabase is None:
            print("üö® ERROR: Supabase client is not initialized!")
            return Div(Alert("‚ùå Internal error: Database not connected!", cls=AlertT.error))

        # ‚úÖ Attempt login
        login_response = supabase.auth.sign_in_with_password({
            "email": email,
            "password": password
        })

        print("üîç Step 2: Login Response:", login_response)

        # ‚úÖ Check if Supabase returned any response
        if login_response is None:
            print("üö® ERROR: Supabase returned None for login!")
            return Div(Alert("‚ùå Supabase error: No response from authentication.", cls=AlertT.error))

        # ‚úÖ Ensure response contains a user
        user = getattr(login_response, "user", None)
        if user is None:
            print("üö® ERROR: Supabase login response is missing user data!")
            return Div(Alert("‚ùå Invalid email or password!", cls=AlertT.error))

        print("üü¢ Step 3: User authenticated successfully:", user.email)

        # ‚úÖ Extract session and access token
        session = getattr(login_response, "session", None)
        if session is None:
            print("üö® ERROR: No session returned from Supabase!")
            return Div(Alert("‚ùå Error: No session found!", cls=AlertT.error))

        access_token = getattr(session, "access_token", None)
        if not access_token:
            print("üö® ERROR: No access token returned!")
            return Div(Alert("‚ùå Error: Access token not found!", cls=AlertT.error))

        print("üü¢ Step 4: Access token retrieved.")

        # ‚úÖ Show success Alert & redirect after 2 seconds
        response = Response(
            Div(
                Alert("‚úÖ Login successful! Redirecting...", cls=AlertT.success),
                Script(f"""
                    setTimeout(function() {{
                        document.cookie = 'auth_token={access_token}; path=/; Secure;';
                        window.location.href='/user';
                    }}, 2000);
                """)
            ),
            status_code=200
        )

        print("üü¢ Step 5: Response sent successfully.")
        return response

    except Exception as e:
        print("‚ùå Exception in signin_account:", e)
        return Div(Alert(f"‚ùå Error: {str(e)}", cls=AlertT.error))
